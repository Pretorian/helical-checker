const video = document.getElementById('video'); const canvas = document.getElementById('canvas'); const ctx = canvas.getContext('2d'); const hud = document.getElementById('hud'); let step = 0; let vertex = null; let baseline = null; let currentAngle = 0; // 1. Better Camera Initialization async function startCamera() { const constraints = { video: { facingMode: "environment" }, audio: false }; try { const stream = await navigator.mediaDevices.getUserMedia(constraints); video.srcObject = stream; video.setAttribute("playsinline", true); // Required for iOS Safari video.play(); } catch (err) { hud.innerText = "Error: Please allow camera access."; console.error(err); } } function resize() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; } // 2. The Logic Fix: Capturing the point function handleAction() { // We use the exact center of the screen as our pointer const centerX = window.innerWidth / 2; const centerY = window.innerHeight / 2; if (step === 0) { vertex = { x: centerX, y: centerY }; step = 1; hud.innerHTML = "Step 1: Vertex Set ✅
Aim at 0° line & tap Next"; } else if (step === 1) { baseline = { x: centerX, y: centerY }; step = 2; hud.style.color = "#4CAF50"; // Green for active measuring } } function reset() { step = 0; vertex = null; baseline = null; hud.style.color = "white"; hud.innerText = "Aim at Corner & tap Next"; } function takeScreenshot() { const tempCanvas = document.createElement('canvas'); tempCanvas.width = canvas.width; tempCanvas.height = canvas.height; const tempCtx = tempCanvas.getContext('2d'); tempCtx.drawImage(video, 0, 0, canvas.width, canvas.height); tempCtx.drawImage(canvas, 0, 0); const link = document.createElement('a'); link.download = `angle-${Math.round(currentAngle)}deg.png`; link.href = tempCanvas.toDataURL('image/png'); link.click(); } // 3. The Drawing Loop (The "Engine") function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); const centerX = window.innerWidth / 2; const centerY = window.innerHeight / 2; if (vertex) { // Draw Vertex Dot ctx.fillStyle = "#ff0000"; ctx.beginPath(); ctx.arc(vertex.x, vertex.y, 10, 0, Math.PI * 2); ctx.fill(); // Draw Line from Vertex to Center (The "Measuring" line) ctx.strokeStyle = "#ffff00"; ctx.lineWidth = 4; ctx.beginPath(); ctx.moveTo(vertex.x, vertex.y); ctx.lineTo(centerX, centerY); ctx.stroke(); if (baseline) { // Draw the fixed Baseline ctx.strokeStyle = "white"; ctx.setLineDash([10, 5]); ctx.beginPath(); ctx.moveTo(vertex.x, vertex.y); ctx.lineTo(baseline.x, baseline.y); ctx.stroke(); ctx.setLineDash([]); // MATH: Calculate angle between baseline and current center const angleBase = Math.atan2(baseline.y - vertex.y, baseline.x - vertex.x); const angleCurrent = Math.atan2(centerY - vertex.y, centerX - vertex.x); let diff = Math.abs((angleCurrent - angleBase) * 180 / Math.PI); if (diff > 180) diff = 360 - diff; currentAngle = diff; hud.innerText = diff.toFixed(1) + "°"; // Draw visual Protractor Arc ctx.beginPath(); ctx.strokeStyle = "rgba(0, 255, 0, 0.4)"; ctx.lineWidth = 30; ctx.arc(vertex.x, vertex.y, 80, angleBase, angleCurrent, (angleCurrent < angleBase)); ctx.stroke(); // Text on canvas for screenshot ctx.fillStyle = "white"; ctx.font = "bold 30px Arial"; ctx.fillText(diff.toFixed(1) + "°", vertex.x + 20, vertex.y - 20); } } requestAnimationFrame(draw); } // Start everything window.addEventListener('resize', resize); resize(); startCamera(); draw();
